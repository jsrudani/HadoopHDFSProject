018    package org.apache.hadoop.hdfs.server.blockmanagement;
019    
020    /**
021     * A immutable object that stores the number of live replicas and
022     * the number of decommissined Replicas.
023     */
024    public class NumberReplicas {
025      private int liveReplicas;
026      private int decommissionedReplicas;
027      private int corruptReplicas;
028      private int excessReplicas;
029      private int replicasOnStaleNodes;
030    
031      NumberReplicas() {
032        initialize(0, 0, 0, 0, 0);
033      }
034    
035      NumberReplicas(int live, int decommissioned, int corrupt, int excess, int stale) {
036        initialize(live, decommissioned, corrupt, excess, stale);
037      }
038    
039      void initialize(int live, int decommissioned, int corrupt, int excess, int stale) {
040        liveReplicas = live;
041        decommissionedReplicas = decommissioned;
042        corruptReplicas = corrupt;
043        excessReplicas = excess;
044        replicasOnStaleNodes = stale;
045      }
046    
047      public int liveReplicas() {
048        return liveReplicas;
049      }
050      public int decommissionedReplicas() {
051        return decommissionedReplicas;
052      }
053      public int corruptReplicas() {
054        return corruptReplicas;
055      }
056      public int excessReplicas() {
057        return excessReplicas;
058      }
059      
060      /**
061       * @return the number of replicas which are on stale nodes.
062       * This is not mutually exclusive with the other counts -- ie a
063       * replica may count as both "live" and "stale".
064       */
065      public int replicasOnStaleNodes() {
066        return replicasOnStaleNodes;
067      }
068    }
