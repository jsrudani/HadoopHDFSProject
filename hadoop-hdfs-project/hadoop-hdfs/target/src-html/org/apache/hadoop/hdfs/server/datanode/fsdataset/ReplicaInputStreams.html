018    package org.apache.hadoop.hdfs.server.datanode.fsdataset;
019    
020    import java.io.Closeable;
021    import java.io.FileDescriptor;
022    import java.io.FileInputStream;
023    import java.io.InputStream;
024    
025    import org.apache.hadoop.io.IOUtils;
026    
027    /**
028     * Contains the input streams for the data and checksum of a replica.
029     */
030    public class ReplicaInputStreams implements Closeable {
031      private final InputStream dataIn;
032      private final InputStream checksumIn;
033    
034      /** Create an object with a data input stream and a checksum input stream. */
035      public ReplicaInputStreams(FileDescriptor dataFd, FileDescriptor checksumFd) {
036        this.dataIn = new FileInputStream(dataFd);
037        this.checksumIn = new FileInputStream(checksumFd);
038      }
039    
040      /** @return the data input stream. */
041      public InputStream getDataIn() {
042        return dataIn;
043      }
044    
045      /** @return the checksum input stream. */
046      public InputStream getChecksumIn() {
047        return checksumIn;
048      }
049    
050      @Override
051      public void close() {
052        IOUtils.closeStream(dataIn);
053        IOUtils.closeStream(checksumIn);
054      }
055    }
